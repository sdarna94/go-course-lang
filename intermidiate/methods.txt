package main

import "fmt"

type rectangle struct {
	width  int
	height int
	length int
}

func (r rectangle) area() int {
	return r.width * r.height * r.length

}

type myint int

	//fmt.Println("Hello from myint")
	return ("Hello from myint")
}

	r.width *= factor  //r.width = r.width * 2
	r.height *= factor //r.height = r.height * 2
	//r.length *= factor
	return r.area()

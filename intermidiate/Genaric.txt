package main

import "fmt"

type stack[T any] struct { // generic struct with type parameter T of any type
	elements []T // generic type slice its a list of elements of type T
	// elements will hold the stack elements

}

func (s *stack[T]) push(elements T) { // method to push element of type T onto the stack
	s.elements = append(s.elements, elements) // append element to the elements slice
}

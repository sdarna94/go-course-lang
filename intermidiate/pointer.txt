package main

import "fmt"

func main() {
	fmt.Println("Hello, World!")

	var a int = 10
	fmt.Println("Value of a:", a)

	var p *int = &a
	fmt.Println("Address of a:", p)
	fmt.Println("Value at address p:", *p)

	modify(p)
	fmt.Println("Modified value of a:", *p)

}

func modify(p *int) {
	//var a int = 20
	fmt.Println("Value inside modify function before increment:", *p)
	//fmt.Println("Value of a:", a)
	*p++
	fmt.Println("Value inside modify function:", *p)
}

package main

import (
	"fmt"
	"sync"
	//"time"
)

func inc(id int, wg *sync.WaitGroup) {
	defer wg.Done()
	fmt.Println("Incrementing ", id)

}

func main() {
	//fmt.Println("Hello, World!")
	var wg sync.WaitGroup

	for i := 0; i < 10; i++ {
		wg.Add(1)
		go inc(i, &wg)
	}

	//time.Sleep(1 * time.Second)
	wg.Wait()
}

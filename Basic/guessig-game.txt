package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	// Seed the random number generator
	// with the current time to ensure different results each run
	source := rand.NewSource(time.Now().UnixNano()) // Use current time as seed
	r := rand.New(source)// Create a new random number generator // with the seeded source
	

	// Generate a random number between 1 and 100
	target := r.Intn(100) + 1 // Intn(100) generates a number from 0 to 99, so we add 1 to shift the range to 1-100
	

	// welcome message
	fmt.Println("Welcome to the Guessing Game!")
	fmt.Println("I have selected a number between 1 and 100. Can you guess it?")

	var guess int   // Variable to store the user's guess

	// Game loop
	for {
		fmt.Print("Enter your guess: ")  // Prompt the user for input
		fmt.Scanln(&guess) // Read the user's input
		// if _, err := fmt.Scanln(&guess); err != nil {
		// 	fmt.Println("Invalid input. Please enter a whole number.")
		// 	continue
		// }

		// Check the guess
		if guess == target {
			fmt.Println("Congratulations! You guessed the correct number:", target)
			break
		} else if guess < target {
			fmt.Println("Too low! Try again.")
		} else {
			fmt.Println("Too high! Try again.")
		}
	}

}
